Classes

class Button {
  constructor (label='Home') {
    this.label = label
  }
  sayLabel () {
    console.log(' -> ', this.label)
  }
}
class BackButton extends Button {
  constructor () {
    super('Back')
  }
}
let btn = new Button()
let backBtn = new BackButton()
btn.sayLabel()     // -> Home
backBtn.sayLabel() // -> Back

Generators

generator.next('Hello')
